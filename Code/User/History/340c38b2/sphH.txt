    known elements:
        let arr: [i32, 5] = {1, 2, 3, 4, 5};
    
    unknown elements:
        let unknown: [i32, 5] = {}; // this will tell the compiler that you have an empty array of 5 32 bit signed ints but dont yet know their values
    
    to set a value, use a for loop
}
